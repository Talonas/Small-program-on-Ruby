327       def load_class(name, kind, option)
328         if name =~ /\A(?:::)?([A-Z]\w*(?:::[A-Z]\w*)*)\z/
329           arg = $2 == "" ? nil : $2
330           [$1, arg]
331         else
332           m = "#{name.inspect} is not a valid class name"
333           @error_stream.puts m
334           raise m
335         end
336         begin
337           eval(name, binding, __FILE__, __LINE__)
338         rescue NameError => e
339           @error_stream.puts "Couldn't find #{kind} class #{name}"
340           @error_stream.puts "Make sure the --require option is specified *before* #{option}"
341           if $_spec_spec ; raise e ; else exit(1) ; end
342         end
343       end
